#include <iostream>
using namespace std;
int main()
{
    int i, j, n;
    cin >> n;
    int l[n][3] = {};
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < 3; j++)
        {
            cin >> l[i][j];
        }
    }
    int count = 0;
    int main_count = 0;
    for (int i = 0; i < n; i++)
    {
        count =0;
        for (int j = 0; j < 3; j++)
        {

            if (l[i][j] == 1)
            {
                count += 1;
            }
        }
        if (count >= 2)
        {
            main_count++;
        }
    }
    cout << main_count << endl;
}
