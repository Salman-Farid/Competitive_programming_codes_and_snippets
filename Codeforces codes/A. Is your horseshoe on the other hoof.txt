#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int i, n = 4, j, count = 0;
    int arr[n];
    for (i = 0; i < n; i++)
    {
        cin >> arr[i];
    }
    sort(arr, arr + n);  
    for (j = 0; j < n; j++)
    {
        if (arr[j] == arr[j+1])
        {
            count += 1;
        }
    }
    cout << count << endl;
}
